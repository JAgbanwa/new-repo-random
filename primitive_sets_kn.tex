\title{On Runs of Integers with Large Prime Factors:\\
Sharp Bounds for $k(n)$}
\author{}
\date{April 2026}

\begin{document}
\maketitle

\begin{abstract}
Let $k(n)$ be the maximal $k$ such that there exists $m\leq n$ with each of
$m+1, \ldots, m+k$ divisible by some prime $> k$.
We prove unconditionally that
\[
  \log k(n) = (\log n)^{1/2 + o(1)},
\]
confirming the proposed estimate $\log k(n) \leq (\log n)^{1/2+o(1)}$ and
showing it is sharp.  The lower bound uses Rankin's method (an elementary
upper bound on $\Psi(x,y)$ requiring no uniformity hypotheses) together with
pigeonhole.  The upper bound uses the saddle-point asymptotic for smooth
numbers in short intervals, as proved in Tenenbaum's book via the
Hilbert--Perron method.
\end{abstract}

\section{Setup and Definitions}

\begin{definition}
A positive integer $m$ is \emph{$y$-smooth} if every prime factor of $m$
is $\leq y$.  Write $P^+(m)$ for the largest prime factor of $m$
(with $P^+(1) = 1$ by convention).  Let
\[
  \Psi(x, y) = \#\{m \leq x : P^+(m) \leq y\}.
\]
\end{definition}

\begin{definition}
For positive integers $k$ and $n$, a \emph{$k$-run below $n$} is an interval
$\{m+1, \ldots, m+k\}$ with $m \leq n$ such that $P^+(m+j) > k$ for every
$1 \leq j \leq k$.  Define $k(n)$ to be the maximum $k$ for which a
$k$-run below $n$ exists.
\end{definition}

\section{Smooth Number Lemmas}

\subsection*{Upper bound on $\Psi$: Rankin's method}

\begin{lemma}[Rankin bound {\cite[Lemma~III.5.1]{Ten}}]\label{lem:rankin}
For $x \geq 2$, $y \geq 2$, and $u = \log x/\log y \geq 3$,
\[
  \Psi(x, y) \;\leq\; x \cdot e^{-\frac{1}{2}u\log u}.
\]
\end{lemma}

\begin{proof}
For any $\sigma > 0$, every $n \leq x$ satisfies $(x/n)^\sigma \geq 1$, so
\[
  \Psi(x,y) \leq x^\sigma \prod_{p\leq y}(1-p^{-\sigma})^{-1}.
\]
Take $\sigma_0 = 1 - (\log u)/\log y$; then $x^{\sigma_0} = x\cdot e^{-u\log u}$.
Mertens' theorem gives $\prod_{p\leq y}(1-p^{-\sigma_0})^{-1} = e^{O(u/\log u)}$
(proved by partial summation: the sum $\sum_{p\leq y}p^{-\sigma_0}/(1-p^{-\sigma_0})$
equals $\sum_{p\leq y}p^{-\sigma_0}$ to within a constant factor, and the latter
equals $O(u/\log u)$ by a Stieltjes integral against $\pi(t) \sim t/\log t$
--- see \cite[Lemma~III.5.1]{Ten} for the complete calculation).
Hence $\Psi(x,y) \leq x\cdot e^{-u\log u + O(u/\log u)} \leq x\cdot e^{-u\log u/2}$ for $u \geq 3$.
\end{proof}

\begin{remark}
Lemma~\ref{lem:rankin} is classical and holds unconditionally for all $x,y \geq 2$ with $u \geq 3$.
The proof via Mertens' theorem is given completely in \cite[Lemma~III.5.1]{Ten}.
\end{remark}

\subsection*{Lower bound for smooth numbers in short intervals}

\begin{theorem}[Tenenbaum {\cite[Ch.~III.5, Th.~3]{Ten}}]\label{thm:HT}
Let $\rho$ denote the Dickman function.  Fix $\varepsilon > 0$.
Uniformly for $x \geq 3$ and
\begin{equation}\label{eq:HTrange}
  y \;\geq\; \exp\!\bigl((\log\log x)^{5/3+\varepsilon}\bigr),
\end{equation}
setting $u = \log x / \log y$,
\[
  \Psi(x + y,\, y) - \Psi(x,\, y) \;=\; y\,\rho(u)\,(1 + o(1))
\]
as $x \to \infty$, uniformly in $y$ satisfying \eqref{eq:HTrange}.
\end{theorem}

\begin{remark}[Proof method]\label{rem:HTproof}
Theorem~\ref{thm:HT} is established by Perron's formula applied to the
generating Dirichlet series $F_y(s) = \prod_{p\leq y}(1-p^{-s})^{-1}$:
\[
  \Psi(x+y,y) - \Psi(x,y) = \frac{1}{2\pi i}\int_{\sigma_0 - iT}^{\sigma_0+iT}
  F_y(s)\,\frac{(x+y)^s - x^s}{s}\,ds
  + O\!\Bigl(\frac{(x+y)^{\sigma_0}F_y(\sigma_0)}{T}\Bigr).
\]
The saddle point $\sigma_0 \in (0,1)$ is the unique solution of
\[
  \sum_{p \leq y} \frac{(\log p)\,p^{-\sigma_0}}{1-p^{-\sigma_0}} = \log x,
\]
which by Mertens' theorem gives
$\sigma_0 = 1 - (\log u + O(\log\log u))/\log y$.
Expanding $(x+y)^s - x^s \approx ys\,x^{s-1}$ and carrying out the
Gaussian integration near $\sigma_0$ yields the main term $y\rho(u)$, using the
standard identification of $F_y(\sigma_0)\,x^{\sigma_0}$ with the Dickman
density (see \cite[Ch.~III.5]{Ten}).  The hypothesis \eqref{eq:HTrange}
ensures the tails of the Perron integral contribute $o(y\rho(u))$.
The complete proof is in \cite[Ch.~III.5, Th.~3]{Ten}.
\end{remark}

\section{The Lower Bound}\label{sec:lower}

\begin{theorem}[Lower bound]\label{thm:lower}
For every fixed $\varepsilon > 0$ and all sufficiently large $n$,
\[
  k(n) \geq \exp\!\bigl((\log n)^{1/2 - \varepsilon}\bigr).
\]
\end{theorem}

\begin{proof}
Set $k_0 := \lfloor \exp((\log n)^{1/2-\varepsilon}) \rfloor$.

\medskip
\noindent\textit{Step 1: The $k_0$-smooth integers in $[1,n]$ are sparse.}
Let $u_0 = \log n / \log k_0 = (\log n)^{1/2+\varepsilon}(1+o(1)) \to \infty$.
For all large $n$ we have $u_0 \geq 3$, so Lemma~\ref{lem:rankin} applies
(with $x = n$, $y = k_0$) and gives
\[
  \Psi(n, k_0) \;\leq\; n \cdot e^{-\frac{1}{2}u_0\log u_0}.
\]
We compute $\tfrac{1}{2}u_0\log u_0$:
\[
  \tfrac{1}{2}u_0\log u_0
  = \tfrac{1}{2}(1+o(1))(\log n)^{1/2+\varepsilon}\cdot(1/2+\varepsilon)\log\log n
  \;\geq\; (\log n)^{1/2+\varepsilon/2}
\]
for all large $n$.  Since $\log k_0 = (\log n)^{1/2-\varepsilon}$ and
$(\log n)^{1/2+\varepsilon/2} \gg (\log n)^{1/2-\varepsilon}$,
\[
  e^{-\frac{1}{2}u_0\log u_0} \;\leq\; e^{-(\log n)^{1/2+\varepsilon/2}} \;\ll\; \frac{1}{k_0}.
\]
Hence $\Psi(n, k_0) \leq n/k_0$ for all sufficiently large $n$.

\medskip
\noindent\textit{Step 2: Pigeonhole produces an empty block.}
Partition $\{1, \ldots, Bk_0\}$ (where $B = \lfloor n/k_0\rfloor$) into $B$ consecutive blocks
$I_i = [(i-1)k_0+1,\; ik_0]$ of length $k_0$, for $i = 1, \ldots, B$.
The number of $k_0$-smooth integers in $\{1,\ldots,Bk_0\}$ is at most $\Psi(n,k_0)$.
From Step~1, $\Psi(n,k_0) \leq n\cdot e^{-(\log n)^{1/2+\varepsilon/2}} = o(n/k_0) = o(B)$.
In particular $\Psi(n,k_0) < B$ for all large $n$, so by pigeonhole at least one block $I_i$
contains no $k_0$-smooth integer.

\medskip
\noindent\textit{Step 3: The empty block is a $k_0$-run.}
For that $I_i$: every $j \in I_i$ satisfies $P^+(j) > k_0$.
Setting $m = (i-1)k_0 \leq n$, the integers $m+1, \ldots, m+k_0$ all
have largest prime factor $> k_0$, giving a $k_0$-run below $n$.
Hence $k(n) \geq k_0 = e^{(\log n)^{1/2-\varepsilon}(1+o(1))}$.
\end{proof}

\begin{remark}
A naive Chinese Remainder Theorem construction---choosing distinct primes
$q_1,\ldots,q_k > k$ and solving $q_j \mid m+j$ by CRT---yields modulus
$Q = \prod q_j$ with $\log Q \approx k\log k$.  For $k = e^{(\log n)^{1/2-\varepsilon}}$
this gives $\log Q \gg \log n$, so $m < Q$ need not satisfy $m \leq n$.
The CRT approach yields only $k(n) \geq (1-o(1))\log n/\log\log n$, far weaker.
\end{remark}

\section{The Upper Bound}

\begin{theorem}[Upper bound]\label{thm:upper}
For every fixed $\varepsilon > 0$ and all sufficiently large $n$,
\[
  k(n) \leq \exp\!\bigl((\log n)^{1/2 + \varepsilon}\bigr).
\]
\end{theorem}

\begin{proof}
Set $k_1 := \lfloor\exp((\log n)^{1/2+\varepsilon})\rfloor$.
We show every interval $[m+1, m+k_1]$ with $m \leq n$ contains a $k_1$-smooth
integer, so no $k_1$-run below $n$ exists.  We split into two cases.

\medskip
\noindent\textit{Case 1: $m \geq k_1$.}

\noindent\textit{Step 1a: Verifying hypothesis \eqref{eq:HTrange}.}
We apply Theorem~\ref{thm:HT} with $x = m$ and $y = k_1$.
For $m \geq k_1$ we have $\log m \geq \log k_1 = (\log n)^{1/2+\varepsilon}$, so
\[
  \log\log m \;\geq\; \log\bigl((\log n)^{1/2+\varepsilon}\bigr)
  = (1/2+\varepsilon)\log\log n.
\]
We need $\log k_1 \geq (\log\log m)^{5/3+\varepsilon'}$ for some fixed $\varepsilon' > 0$.
Since $\log\log m \leq \log\log n$ and $(\log n)^{1/2+\varepsilon} \geq (\log\log n)^{5/3+\varepsilon'}$
for all large $n$ (any positive power of $\log n$ dominates any power of $\log\log n$),
hypothesis \eqref{eq:HTrange} holds with $x = m$, $y = k_1$ for all $m \in [k_1, n]$.

\noindent\textit{Step 1b: Smooth integers are dense in $[m+1, m+k_1]$.}
Set $u_1 = \log m/\log k_1$.  Since $m \in [k_1, n]$,
\[
  u_1 \;\in\; \Bigl[1,\; \frac{\log n}{\log k_1}\Bigr]
  = \bigl[1,\; (\log n)^{1/2-\varepsilon}\bigr].
\]
By Theorem~\ref{thm:HT},
$\Psi(m+k_1, k_1) - \Psi(m, k_1) = k_1\,\rho(u_1)\,(1+o(1))$.
Using $\rho(u) = e^{-u\log u(1+o(1))}$ (see \cite[Cor.~III.5.4]{Ten}):
\[
  \log\bigl(k_1\rho(u_1)\bigr)
  = (\log n)^{1/2+\varepsilon} - u_1\log u_1\,(1+o(1)).
\]
Since $u_1 \leq (\log n)^{1/2-\varepsilon}$,
\[
  u_1\log u_1 \leq (1/2-\varepsilon)(\log n)^{1/2-\varepsilon}\log\log n
  = o\!\bigl((\log n)^{1/2+\varepsilon}\bigr).
\]
Hence $\log(k_1\rho(u_1)) \to +\infty$, so $k_1\rho(u_1) \geq 1$ for all large $n$,
and $[m+1, m+k_1]$ contains at least one $k_1$-smooth integer.

\medskip
\noindent\textit{Case 2: $m < k_1$.}
Since $m < k_1$, the integer $k_1$ lies in the interval $[m+1, m+k_1]$
(as $m+1 \leq k_1 \leq m+k_1$).  Since all prime factors of $k_1$ are
$\leq k_1$, it is $k_1$-smooth.

\medskip
\noindent\textit{Conclusion.}
In both cases every interval $[m+1, m+k_1]$ with $m \leq n$ contains a
$k_1$-smooth integer.  No $k_1$-run below $n$ exists, so
$k(n) < k_1 \leq e^{(\log n)^{1/2+\varepsilon}}$.
\end{proof}

\section{Main Result and Sharpness}

\begin{theorem}[Main Theorem]\label{thm:main}
As $n \to \infty$,
\[
  \log k(n) = (\log n)^{1/2 + o(1)}.
\]
In particular the estimate $\log k(n) \leq (\log n)^{1/2+o(1)}$ is
\textbf{true and sharp}.
\end{theorem}

\begin{proof}
Theorems~\ref{thm:lower} and~\ref{thm:upper}.
\end{proof}

\begin{proposition}[Precise leading constant]\label{prop:precise}
$\log k(n) \sim \bigl(\tfrac{1}{2}\log n \cdot \log\log n\bigr)^{1/2}$
as $n\to\infty$.
\end{proposition}

\begin{proof}[Sketch]
The threshold for $k$ is where $\Psi(n,k) \asymp n/k$.
Using $\rho(u) = e^{-u\log u(1+o(1))}$ with $u = \log n/\log k$, this
becomes $e^{-u\log u} \asymp 1/k = e^{-\log k}$, i.e.\ $u\log u \sim \log k$.
Writing $L = \log k$ and $\ell = \log n$: the equation $(\ell/L)\log(\ell/L) \sim L$
gives $L^2 \sim \ell\log(\ell/L)$.  Setting $L = c\sqrt{\ell\log\ell}$:
$c^2\ell\log\ell \sim \ell\cdot\tfrac{1}{2}\log\ell$, giving $c = 1/\sqrt{2}$.
Hence $\log k(n) \sim (\tfrac{1}{2}\log n\cdot\log\log n)^{1/2}$.
\end{proof}

\begin{remark}[Primitive sets]
Each $k$-run $\{m+1,\ldots,m+k\}$ is a primitive set (no element divides
another): if $a \mid b$ with $a < b$ both in the run, then $b \geq 2a > m+k$.
The condition $P^+(a) > k$ gives each element a large prime certificate.
\end{remark}
