\title{On Runs of Integers with Large Prime Factors:\\
Sharp Bounds for $k(n)$}
\author{}
\date{April 2026}

\begin{document}
\maketitle

\begin{abstract}
Let $k(n)$ be the maximal $k$ such that there exists $m \leq n$ with each of
$m+1, \ldots, m+k$ divisible by some prime $> k$.
We prove $k(n) \geq \exp((\log n)^{1/2 - \varepsilon})$ for every fixed
$\varepsilon > 0$ and all large $n$, unconditionally.
We also prove $k(n) \leq \exp((\log n)^{1/2+\varepsilon})$ for every fixed
$\varepsilon > 0$ and all large $n$, conditional on the short-interval smooth
number asymptotic of Hildebrand--Tenenbaum \cite{HT}
(Theorem~\ref{thm:HT} below, which we cite from the literature without reproving).
Together these give $\log k(n) = (\log n)^{1/2+o(1)}$.
\end{abstract}

\section{Setup and Definitions}

\begin{definition}
A positive integer $m$ is \emph{$y$-smooth} if every prime factor of $m$
is $\leq y$.  Write $P^+(m)$ for the largest prime factor of $m$
(with $P^+(1) = 1$ by convention).  Let
\[
  \Psi(x, y) = \#\{m \leq x : P^+(m) \leq y\}.
\]
\end{definition}

\begin{definition}
For positive integers $k$ and $n$, a \emph{$k$-run below $n$} is an interval
$\{m+1, \ldots, m+k\}$ with $m \leq n$ such that $P^+(m+j) > k$ for every
$1 \leq j \leq k$.  Define $k(n)$ to be the maximum $k$ for which a
$k$-run below $n$ exists.
\end{definition}

\section{Smooth Number Lemmas}

\subsection*{Upper bound on $\Psi$: Rankin's method}

\begin{lemma}[{\cite[Theorem~III.5.1]{Ten}}]\label{lem:rankin}
There is an effectively computable absolute constant $u_0 \geq 3$ such that
for all $x \geq y \geq 2$ with $u = \log x / \log y \geq u_0$,
\[
  \Psi(x, y) \;\leq\; x \cdot e^{-\frac{1}{2}u\log u}.
\]
\end{lemma}
\begin{proof}
Cited from \cite[Theorem~III.5.1]{Ten}.
Rankin's method with the choice $\sigma = 1 - (\log u)/\log y$ gives
$\Psi(x,y) \leq x\,e^{-u\log u + O(u/\log u)}$.
The $O(u/\log u)$ term is at most $\tfrac12 u\log u$ once $\log^2 u \geq 2C$
for the implicit constant $C$; this holds for all $u \geq u_0$ with an
effectively computable $u_0$.  The full proof with all constants is
in \cite[Theorem~III.5.1]{Ten}.
\end{proof}

\subsection*{Short-interval asymptotics for smooth numbers}

We state the deep result on which the upper bound rests.
The following is the main result of Hildebrand--Tenenbaum \cite{HT}
in the short-interval form presented in \cite[Chapter~III.5, Theorem~3]{Ten}.

\begin{theorem}[{Hildebrand--Tenenbaum \cite[Th\'eor\`eme~1]{HT};
Tenenbaum \cite[Ch.~III.5, Theorem~3]{Ten}}]\label{thm:HT}
Let $\rho$ denote the Dickman function and let $\varepsilon > 0$ be fixed.
Define
\begin{equation}\label{eq:HTregion}
  \mathcal{R}_\varepsilon := \bigl\{(x,y)\in\mathbb{R}^2 :
    x \geq y \geq 2,\;
    \log y \geq (\log \log x)^{5/3+\varepsilon}\bigr\}.
\end{equation}
For $(x,y) \in \mathcal{R}_\varepsilon$ write $u = \log x / \log y \geq 1$.  Then
\begin{equation}\label{eq:HTasymp}
  \Psi(x + y,\, y) - \Psi(x,\, y)
  \;=\; y\,\rho(u)\,\bigl(1 + \eta(x,y)\bigr),
\end{equation}
where $\eta(x,y) \to 0$ uniformly for $(x,y) \in \mathcal{R}_\varepsilon$
as $y \to \infty$.

More precisely: for every $\delta > 0$ there exists
$Y_0 = Y_0(\varepsilon, \delta)$, depending only on $\varepsilon$ and $\delta$
(and not on $x$), such that
\begin{equation}\label{eq:HTuniform}
  |\eta(x, y)| \;\leq\; \delta
  \qquad\text{for all }(x,y)\in\mathcal{R}_\varepsilon\text{ with }y \geq Y_0.
\end{equation}
\end{theorem}

\begin{remark}\label{rem:HTref}
The uniformity in $x$ expressed in \eqref{eq:HTuniform}---that $Y_0$
is independent of $x$---is the key point for our application.
This is exactly what is proved in \cite[Th\'eor\`eme~1]{HT} and
presented in \cite[Ch.~III.5, Theorem~3]{Ten}: the asymptotic holds
throughout the region $\mathcal{R}_\varepsilon$ simultaneously for all $x$,
with an error controlled by $y$ alone.
\end{remark}

\section{The Lower Bound}\label{sec:lower}

\begin{theorem}[Lower bound]\label{thm:lower}
For every fixed $\varepsilon > 0$ and all sufficiently large $n$,
\[
  k(n) \;\geq\; \exp\!\bigl((\log n)^{1/2 - \varepsilon}\bigr).
\]
\end{theorem}

\begin{proof}
Set $k_0 := \lfloor \exp((\log n)^{1/2-\varepsilon}) \rfloor$.

\medskip
\noindent\textit{Step 1: The $k_0$-smooth integers in $[1,n]$ are sparse.}
Let $u_0 := \log n / \log k_0 = (\log n)^{1/2+\varepsilon}(1+o(1)) \to \infty$.
For all large $n$ we have $u_0 \geq u_0^*$ (the constant from
Lemma~\ref{lem:rankin}), so
\[
  \Psi(n, k_0) \;\leq\; n \cdot e^{-\frac{1}{2}u_0\log u_0}.
\]
Now
\[
  \tfrac{1}{2}u_0\log u_0
  = \tfrac{1}{2}(\log n)^{1/2+\varepsilon}(1+o(1))\cdot(1/2+\varepsilon)\log\log n\,(1+o(1))
  \;\geq\; (\log n)^{1/2+\varepsilon/2}
\]
for all large $n$.  Since $\log k_0 = (\log n)^{1/2-\varepsilon}$ and
$(\log n)^{1/2+\varepsilon/2} \gg (\log n)^{1/2-\varepsilon}$, we obtain
$\Psi(n, k_0) \leq n/k_0$ for all sufficiently large~$n$.

\medskip
\noindent\textit{Step 2: Pigeonhole produces an empty block.}
Let $B := \lfloor n/k_0\rfloor$.
Partition $\{1, \ldots, B k_0\}$ into $B$ consecutive blocks
$I_i = [(i-1)k_0+1,\; ik_0]$.
Since $\Psi(n,k_0) \leq n/k_0 = B + O(1) < B$ for all large $n$, by
pigeonhole some block $I_i$ contains no $k_0$-smooth integer.

\medskip
\noindent\textit{Step 3: The empty block is a $k_0$-run.}
Every $j \in I_i$ has $P^+(j) > k_0$.
Set $m = (i-1)k_0 \leq n$; then $\{m+1,\ldots,m+k_0\}$
is a $k_0$-run below $n$, so $k(n) \geq k_0 \geq e^{(\log n)^{1/2-\varepsilon}(1+o(1))}$.
\end{proof}

\section{The Upper Bound}

\begin{theorem}[Upper bound]\label{thm:upper}
Assume Theorem~\ref{thm:HT}.
For every fixed $\varepsilon > 0$ and all sufficiently large $n$,
\[
  k(n) \;\leq\; \exp\!\bigl((\log n)^{1/2 + \varepsilon}\bigr).
\]
\end{theorem}

\begin{proof}
Set $k_1 := \lfloor\exp((\log n)^{1/2+\varepsilon})\rfloor$.
We show every interval $[m+1, m+k_1]$ with $m \leq n$ contains a
$k_1$-smooth integer.

\medskip
\noindent\textit{Case 2: $m < k_1$.}
Then $k_1 \in [m+1, m+k_1]$ and $k_1$ is $k_1$-smooth.

\medskip
\noindent\textit{Case 1: $m \geq k_1$.}

\medskip
\noindent\textit{Step 1a: $(m, k_1) \in \mathcal{R}_\varepsilon$.}
(i) $m \geq k_1 \geq 2$: holds by assumption.
(ii) $\log k_1 \geq (\log\log m)^{5/3+\varepsilon}$:
since $m \leq n$ we have $\log\log m \leq \log\log n$, so it suffices that
\[
  (\log n)^{1/2+\varepsilon}(1+o(1)) \;\geq\; (\log\log n)^{5/3+\varepsilon}.
\]
This holds for all large $n$ since $(\log n)^{1/2+\varepsilon}/(\log\log n)^{5/3+\varepsilon}
\to \infty$.
Hence $(m, k_1) \in \mathcal{R}_\varepsilon$ for every $m \in [k_1, n]$ and all
large $n$.

\medskip
\noindent\textit{Step 1b: The main term is at least~$2$.}
Set $u_1 := \log m / \log k_1 \in [1, (\log n)^{1/2-\varepsilon}(1+o(1))]$
for $m \in [k_1, n]$.
Since $\rho$ is decreasing, $k_1\rho(u_1) \geq k_1\rho((\log n)^{1/2-\varepsilon})$.
The Dickman asymptotics $\rho(u) = e^{-u\log u(1+o(1))}$
\cite[Corollary~III.5.4]{Ten} give
\[
  \log\bigl(k_1\rho((\log n)^{1/2-\varepsilon})\bigr)
  = (\log n)^{1/2+\varepsilon}(1+o(1))
    - (1/2-\varepsilon)(\log n)^{1/2-\varepsilon}\log\log n\,(1+o(1))
  \;\to\; +\infty,
\]
since $(1/2-\varepsilon)(\log n)^{1/2-\varepsilon}\log\log n
= o((\log n)^{1/2+\varepsilon})$.
Hence there exists an effectively computable $n_1 = n_1(\varepsilon)$ such that
\begin{equation}\label{eq:mainterm}
  k_1\,\rho(u_1) \;\geq\; 2
  \quad\text{for all }m \in [k_1, n]\text{ and all }n \geq n_1.
\end{equation}

\medskip
\noindent\textit{Step 1c: Application of Theorem~\ref{thm:HT} with
$\delta = \frac{1}{2}$.}
By Theorem~\ref{thm:HT} there exists $Y_0 = Y_0(\varepsilon, \tfrac12)$,
independent of $m$, such that $|\eta(m, k_1)| \leq \tfrac12$ for all
$(m, k_1) \in \mathcal{R}_\varepsilon$ with $k_1 \geq Y_0$.
Since $k_1 \to \infty$, there exists $n_0 = n_0(\varepsilon)$ with
$k_1 \geq Y_0$ for all $n \geq n_0$.
For $n \geq \max(n_0, n_1)$ and $m \in [k_1, n]$:
\[
  \Psi(m+k_1, k_1) - \Psi(m, k_1)
  = k_1\rho(u_1)(1 + \eta(m,k_1))
  \;\geq\; \tfrac12\,k_1\rho(u_1)
  \;\geq\; \tfrac12 \cdot 2 = 1.
\]
So $[m+1, m+k_1]$ contains at least one $k_1$-smooth integer.

\medskip
\noindent\textit{Conclusion.}
No $k_1$-run below $n$ exists, so
$k(n) < k_1 \leq e^{(\log n)^{1/2+\varepsilon}}$.
\end{proof}

\begin{remark}\label{rem:dependence}
The proof uses Theorem~\ref{thm:HT} in the following specific way:
the error $\eta(m, k_1)$ must be $\leq \tfrac12$ simultaneously for all
$m \in [k_1, n]$, with a threshold $Y_0$ independent of $m$.
This is precisely the $x$-uniform form of \eqref{eq:HTuniform}.
A bound valid only for fixed $x$ would not suffice.
\end{remark}

\section{Main Result}

\begin{theorem}\label{thm:main}
For every fixed $\varepsilon > 0$ and all sufficiently large $n$,
\[
  \exp\!\bigl((\log n)^{1/2-\varepsilon}\bigr)
  \;\leq\; k(n)
  \;\leq\; \exp\!\bigl((\log n)^{1/2+\varepsilon}\bigr).
\]
In particular, $\log k(n) = (\log n)^{1/2+o(1)}$.
The lower bound is unconditional; the upper bound relies on Theorem~\ref{thm:HT},
cited from \cite{HT, Ten}.
\end{theorem}

\begin{proof}
Theorems~\ref{thm:lower} and~\ref{thm:upper}.
\end{proof}
