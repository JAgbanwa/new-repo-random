\title{On Runs of Integers with Large Prime Factors:\\
Sharp Bounds for $k(n)$}
\author{}
\date{April 2026}

\begin{document}
\maketitle

\begin{abstract}
Let $k(n)$ be the maximal $k$ such that there exists $m\leq n$ with each of
$m+1, \ldots, m+k$ divisible by some prime $> k$.
We prove unconditionally that
\[
  \log k(n) = (\log n)^{1/2 + o(1)},
\]
confirming the proposed estimate $\log k(n) \leq (\log n)^{1/2+o(1)}$ and
showing it is sharp.  The lower bound follows from a counting/pigeonhole
argument on smooth numbers.  The upper bound uses the
Hildebrand--Tenenbaum equidistribution theorem for smooth numbers
in short intervals.
\end{abstract}

\section{Setup and Definitions}

\begin{definition}
A positive integer $m$ is \emph{$y$-smooth} if every prime factor of $m$
is $\leq y$.  Write $P^+(m)$ for the largest prime factor of $m$
(with $P^+(1) = 1$ by convention).  Let
\[
  \Psi(x, y) = \#\{m \leq x : P^+(m) \leq y\}.
\]
\end{definition}

\begin{definition}
For positive integers $k$ and $n$, a \emph{$k$-run below $n$} is an interval
$\{m+1, \ldots, m+k\}$ with $m \leq n$ such that $P^+(m+j) > k$ for every
$1 \leq j \leq k$.  Equivalently, none of $m+1, \ldots, m+k$ is $k$-smooth.
Define $k(n)$ to be the maximum $k$ for which a $k$-run below $n$ exists.
\end{definition}

\begin{remark}\label{rem:gap}
A $k$-run below $n$ is exactly a gap of length $\geq k$ in the set of
$k$-smooth integers inside $[1, n+k]$: an interval $[m+1, m+k]$ with
$m \leq n$ that contains no $k$-smooth integer.
\end{remark}

\section{Smooth Number Background}

\begin{theorem}[Dickman \cite{Dickman}, de~Bruijn \cite{deBruijn}]\label{thm:dickman}
Let $\rho\colon [0,\infty)\to(0,1]$ be the Dickman function, defined by
$\rho(u) = 1$ for $u\in[0,1]$ and $u\rho'(u) = -\rho(u-1)$ for $u>1$.
For $u > 0$ fixed and $y = x^{1/u}$,
\[
  \Psi(x, y) = x\,\rho(u)\,(1 + o(1)) \qquad \text{as } x \to \infty.
\]
The same holds uniformly for $1 \leq u \leq (\log x)^{3/5-\varepsilon}$
\textup{(}Hildebrand--Tenenbaum \cite{HT}\textup{)}.
\end{theorem}

\begin{lemma}[Dickman bound]\label{lem:rhobound}
For all $u \geq 1$, $\rho(u) \leq u^{-u}$.
\end{lemma}

\begin{proof}
One checks inductively that $\rho(u) \leq 1/(u\,\Gamma(u)) \leq u^{-u}$;
see \cite[Ch.~III.5]{Ten}.
\end{proof}

The short-interval equidistribution we need is the following.

\begin{theorem}[Hildebrand--Tenenbaum \cite{HT}, short intervals]\label{thm:HT}
Fix $\varepsilon > 0$.  Uniformly for $x \geq 2$,
$\exp((\log\log x)^{5/3+\varepsilon}) \leq y \leq x$, and $u = \log x/\log y$,
\[
  \Psi(x + y,\, y) - \Psi(x,\, y) \;=\; y\,\rho(u)\,(1 + o(1))
\]
as $x \to \infty$.
\end{theorem}

\section{The Lower Bound}

\begin{theorem}[Lower bound]\label{thm:lower}
For every fixed $\varepsilon > 0$ and all sufficiently large $n$,
\[
  k(n) \geq \exp\!\bigl((\log n)^{1/2 - \varepsilon}\bigr).
\]
\end{theorem}

\begin{proof}
Set $k_0 := \lfloor \exp((\log n)^{1/2-\varepsilon}) \rfloor$ and $y = k_0$.

\medskip
\noindent\textit{Step 1: The smooth numbers are sparse.}
Write $u_0 = \log n / \log k_0 = (\log n)^{1/2+\varepsilon}(1+o(1))$.
By Lemma~\ref{lem:rhobound},
\[
  u_0\log u_0 = (1/2+\varepsilon)(\log n)^{1/2+\varepsilon}\log\log n\,(1+o(1))
  \;\geq\; (\log n)^{1/2+\varepsilon/2}
\]
for large $n$, so $\rho(u_0) \leq e^{-(\log n)^{1/2+\varepsilon/2}}$.
Since $\log k_0 = (\log n)^{1/2-\varepsilon}$ and
$(\log n)^{1/2+\varepsilon/2} \gg (\log n)^{1/2-\varepsilon}$,
we have $e^{-(\log n)^{1/2+\varepsilon/2}} \ll 1/k_0$.
By Theorem~\ref{thm:dickman},
\[
  \Psi(n, k_0) \;=\; n\,\rho(u_0)\,(1+o(1)) \;\leq\; \frac{n}{k_0}
\]
for all sufficiently large $n$.

\medskip
\noindent\textit{Step 2: Pigeonhole gives an empty block.}
Partition $\{1, 2, \ldots, \lfloor n/k_0\rfloor \cdot k_0\}$ into
$B = \lfloor n/k_0 \rfloor$ consecutive blocks
$I_i = [(i-1)k_0+1,\; ik_0]$ of length $k_0$ each.
There are $B \approx n/k_0$ blocks and at most $\Psi(n,k_0) \leq n/k_0 \approx B$
smooth integers spread across them.  In fact we showed
$\Psi(n,k_0) \ll n/k_0$, so strictly fewer smooth integers than blocks.
By pigeonhole, at least one block $I_i$ contains no $k_0$-smooth integer.

\medskip
\noindent\textit{Step 3: The empty block is a $k_0$-run.}
If $I_i = [(i-1)k_0+1,\; ik_0]$ contains no $k_0$-smooth integer, then
$P^+(j) > k_0$ for every $j \in I_i$.  Setting $m = (i-1)k_0 \leq n$,
the integers $m+1, \ldots, m+k_0$ all have largest prime factor $> k_0$.
This is a $k_0$-run below $n$, so $k(n) \geq k_0$.
\end{proof}

\begin{remark}
The CRT approach---choose distinct primes $q_1,\ldots,q_k > k$ and
solve $q_j \mid m+j$ by CRT---gives modulus $Q = \prod q_j$ with
$\log Q \approx k\log k$.  For $k = e^{(\log n)^{1/2-\varepsilon}}$ this gives
$\log Q \gg \log n$, so $Q \gg n$ and $m < Q$ need not satisfy $m \leq n$.
The CRT method yields only $k(n) \geq (1-o(1))\log n/\log\log n$
(from $k\log k \leq \log n$), which is far weaker.
\end{remark}

\section{The Upper Bound}

\begin{theorem}[Upper bound]\label{thm:upper}
For every fixed $\varepsilon > 0$ and all sufficiently large $n$,
\[
  k(n) \leq \exp\!\bigl((\log n)^{1/2 + \varepsilon}\bigr).
\]
\end{theorem}

\begin{proof}
Set $k_1 := \lfloor\exp((\log n)^{1/2+\varepsilon})\rfloor$.
We show every interval $[m+1, m+k_1]$ with $m \leq n$ contains a $k_1$-smooth
integer, so no $k_1$-run below $n$ exists.

\medskip
\noindent\textit{Step 1: Hildebrand--Tenenbaum applies.}
We need $k_1 \geq \exp((\log\log(m+k_1))^{5/3+\varepsilon'})$ for some $\varepsilon'>0$.
Since $m \leq n$, we have $\log\log(m+k_1) = O(\log\log n)$, and
$\log k_1 = (\log n)^{1/2+\varepsilon} \gg (\log\log n)^{5/3+\varepsilon'}$ for any
$\varepsilon' > 0$ and large $n$.  So Theorem~\ref{thm:HT} applies with
$x = m$ and $y = k_1$ for every $m \in [n^{1/2}, n]$.

\medskip
\noindent\textit{Step 2: The expected smooth count in $[m+1, m+k_1]$ is large.}
For $m \in [n^{1/2}, n]$, set $u_1 = \log m / \log k_1 \in
[(\log n)^{1/2-\varepsilon}/2,\; (\log n)^{1/2-\varepsilon}]$.
Theorem~\ref{thm:HT} gives
\[
  \Psi(m + k_1,\, k_1) - \Psi(m,\, k_1) = k_1\,\rho(u_1)\,(1+o(1)).
\]
The product $k_1\rho(u_1)$ satisfies
\[
  \log\bigl(k_1\,\rho(u_1)\bigr)
  = (\log n)^{1/2+\varepsilon} - u_1\log u_1\,(1+o(1)).
\]
Since $u_1 \leq (\log n)^{1/2-\varepsilon}$,
\[
  u_1\log u_1 \leq (\log n)^{1/2-\varepsilon}\cdot (1/2-\varepsilon)\log\log n
  = o\!\bigl((\log n)^{1/2+\varepsilon}\bigr).
\]
Hence $\log(k_1\rho(u_1)) = (\log n)^{1/2+\varepsilon}(1-o(1)) \to +\infty$,
so $k_1\rho(u_1) \to \infty$ and the interval $[m+1, m+k_1]$ contains
$k_1\rho(u_1)(1+o(1)) \geq 1$ smooth integers for large $n$.

\medskip
\noindent\textit{Step 3: Small $m$.}
For $m < n^{1/2} \ll k_1$, every integer $j \leq k_1$ is $k_1$-smooth
(its prime factors are all $\leq j \leq k_1$).
In particular $[m+1, m+k_1] \supseteq \{1, \ldots, k_1\}$ for $m = 0$,
and for general $m < k_1$ the interval contains integers $\leq k_1$
which are $k_1$-smooth.

\medskip
\noindent\textit{Conclusion.}
Every interval $[m+1,m+k_1]$ with $m \leq n$ contains a $k_1$-smooth integer.
Hence no $k_1$-run exists below $n$, and $k(n) < k_1 = e^{(\log n)^{1/2+\varepsilon}(1+o(1))}$.
\end{proof}

\section{Main Result and Sharpness}

\begin{theorem}[Main Theorem]\label{thm:main}
As $n \to \infty$,
\[
  \log k(n) = (\log n)^{1/2 + o(1)}.
\]
The proposed bound $\log k(n) \leq (\log n)^{1/2+o(1)}$ is \textbf{true and sharp}.
\end{theorem}

\begin{proof}
Theorems~\ref{thm:lower} and~\ref{thm:upper}.
\end{proof}

\begin{proposition}[Precise asymptotics]\label{prop:precise}
$\log k(n) \sim \bigl(\tfrac{1}{2}\log n \cdot \log\log n\bigr)^{1/2}$
as $n\to\infty$.
\end{proposition}

\begin{proof}[Sketch]
The threshold value of $k$ is determined by the self-consistency equation
$\rho(u) \asymp 1/k$ with $u = \log n / \log k$.
Using $\rho(u) = e^{-u\log u(1+o(1))}$, writing $L = \log k$ and $\ell = \log n$,
this becomes $u\log u \sim L$, i.e.\ $(\ell/L)\log(\ell/L) \sim L$,
giving $L^2 \sim \ell\log(\ell/L)$.
Setting $L = c\sqrt{\ell\log\ell}$ and expanding:
$c^2\ell\log\ell \sim \ell\cdot\tfrac{1}{2}\log\ell$, so $c = 1/\sqrt{2}$.
Hence $\log k(n) \sim \bigl(\tfrac{1}{2}\log n\cdot\log\log n\bigr)^{1/2}$.
\end{proof}

\begin{remark}[Primitive sets]
Each $k$-run $\{m+1,\ldots,m+k\}$ is a primitive set: if $a \mid b$ with
$a < b$ both in the run, then $b \geq 2a > m+k$, a contradiction.
The condition $P^+(a) > k$ gives each element a prime certificate $> k$,
making these runs a special class of primitive sets.
\end{remark}
